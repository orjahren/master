\NewDocumentCommand{\Nref}{s m}{%
  \hyperref[#2]{%
    \nameref*{#2}%
    \IfBooleanF{#1}{%
      \textsuperscript{\ttfamily\tiny\(\,\to\,\)p.\labelcpageref*{#2}}%
    }%
  }%
}

\RenewDocumentCommand{\Cref}{s m}{%
  \hyperref[#2]{%
    \nameCref{#2}~\labelcref*{#2}%
    \IfBooleanF{#1}{%
      \textsuperscript{\ttfamily\tiny\(\,\to\,\)p.\labelcpageref*{#2}}%
    }%
  }%
}
